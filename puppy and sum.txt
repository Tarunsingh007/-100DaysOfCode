#include<bits/stdc++.h> 
int main() 
{
using namespace std; 
    long long int t;
    cin>>t;
    while(t--)
    {
        pair<int,int> p; int sum=0;
        cin>>p.first>>p.second;
        while(p.first--)
        {
            sum=0;
            for(int i=1; i<=p.second; i++)
            {
                sum=sum+i;
            }
            p.second=sum;
        }
        cout<< sum<< endl;
    }
}

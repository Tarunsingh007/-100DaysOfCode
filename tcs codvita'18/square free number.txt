#include<bits/stdc++.h>
#include<vector>
using namespace std;
int main()
{
    int n,c=0;
    cin>>n;
    vector<int> v, p;
    for(int i=2; i<=n; i++)
    {
        if(n%i==0)
        v.push_back(i);
    }
    for(int i=0; i<v.size(); i++)
    {
        if(int(sqrt(v[i]))==double(sqrt(v[i])))
        p.push_back(v[i]);
    }
    for(int i=0; i<p.size(); i++)
    {
        for(int j=0;j<v.size(); j++)
        {
            if(v[j]%p[i]==0)
            {
                v.erase(v.begin()+j);
                j=0;
            }
        }
    }
    cout<<v.size()<<endl;
    return 0;
}
